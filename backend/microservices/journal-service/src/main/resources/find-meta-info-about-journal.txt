db.getCollection('journal').aggregate([
  { $project: {_id: { $toString: "$_id"}}},
  { $lookup: {
       from: 'journal-button',
       localField: 'button_id',
       foreignField: 'id',
       as: 'button_list'
  }},
   {
    $lookup: {
       from: 'journal-column',
       localField: 'column_id',
       foreignField: 'id',
       as: 'column_metadata'
    }
  },
])
